PAGE_SIZE = 20
MAX_PAGE  = 5  # top-100 = 5 pages of 20


def ranking_service(usuario_id: int, role: str, page: int = 1) -> tuple[dict, int]:
    """Retorna pagina do ranking + entrada propria do usuario autenticado.

    Paginacao: PAGE_SIZE=20, MAX_PAGE=5 (top-100 total).
    own_entry: sempre incluido se o usuario for aluno; null caso contrario.
    Seguranca: get_ranking_page e get_own_rank usam ROW_NUMBER() - apenas PostgreSQL.
    """
    if page < 1:
        page = 1
    if page > MAX_PAGE:
        page = MAX_PAGE

    offset = (page - 1) * PAGE_SIZE

    try:
        rows = get_ranking_page(limite=PAGE_SIZE, offset=offset)
        ranking = [
            {
                "posicao": int(r["posicao"]),
                "id":      r["id"],
                "nome":    r["name"],
                "xp":      int(r["xp"]),
            }
            for r in rows
        ]

        own_entry = None
        if role == "aluno":
            proprio = get_own_rank(usuario_id)
            if proprio:
                own_entry = {
                    "posicao": proprio["posicao"],
                    "id":      usuario_id,
                    "nome":    proprio["name"],
                    "xp":      proprio["xp"],
                }

        return {
            "ranking":   ranking,
            "own_entry": own_entry,
            "page":      page,
            "page_size": PAGE_SIZE,
            "has_more":  len(rows) == PAGE_SIZE and page < MAX_PAGE,
        }, 200

    except Exception as e:
        return {"error": f"erro ao buscar ranking: {e}"}, 500
```

@bp.route('/ranking', methods=['GET'])
@jwt_required()
def get_ranking():
    """Retorna ranking paginado de alunos por XP.

    Query params: page (int, default 1) — paginas de 20 itens (top-100 = 5 paginas).
    Resposta inclui own_entry com posicao do usuario autenticado (se for aluno).
    Seguranca: requer JWT valido. Qualquer role pode visualizar o ranking.
    """
    usuario_id = get_jwt_identity()
    claims     = get_jwt()
    role       = claims.get("role", "")
    try:
        page = int(request.args.get("page", 1))
    except (TypeError, ValueError):
        page = 1

    resposta, status = ranking_service(usuario_id, role, page)
    return jsonify(resposta), status
```
