```typescript
import React, { useCallback, useState } from 'react';
import {
  FlatList,
  ActivityIndicator,
  RefreshControl,
  View,
} from 'react-native';
import { useFocusEffect, Stack } from 'expo-router';
import { YStack, XStack, Text } from 'tamagui';
import { SafeAreaView } from 'react-native-safe-area-context';
import { palette } from 'app/constants/style';
import { useAuth } from 'app/context/AuthContext';
import api from 'app/services/api';
import { AppButton } from 'app/components/AppButton';

interface RankingEntry {
  posicao: number;
  id: number;
  nome: string;
  xp: number;
}

interface RankingResponse {
  ranking: RankingEntry[];
  own_entry: RankingEntry | null;
  page: number;
  page_size: number;
  has_more: boolean;
}

function RankingRow({ item, isOwn = false }: { item: RankingEntry; isOwn?: boolean }) {
  return (
    <XStack
      height={52}
      px="$4"
      ai="center"
      jc="space-between"
      borderBottomWidth={1}
      borderColor="rgba(255,255,255,0.08)"
      backgroundColor={isOwn ? 'rgba(255,215,0,0.15)' : 'transparent'}
      borderLeftWidth={isOwn ? 3 : 0}
      borderLeftColor={isOwn ? '#FFD700' : 'transparent'}
    >
      <XStack gap="$3" ai="center">
        <Text fontSize={15} fontWeight="bold" color="white" width={28}>
          #{item.posicao}
        </Text>
        <Text fontSize={15} color="white">
          {item.nome}
        </Text>
      </XStack>
      <Text fontSize={15} fontWeight="bold" color={palette.primaryGreen}>
        {item.xp} XP
      </Text>
    </XStack>
  );
}

export default function RankingScreen() {
  const { role } = useAuth();

  const [ranking, setRanking]       = useState<RankingEntry[]>([]);
  const [propria, setPropria]       = useState<RankingEntry | null>(null);
  const [pagina, setPagina]         = useState(1);
  const [hasMore, setHasMore]       = useState(false);
  const [carregando, setCarregando] = useState(true);
  const [atualizando, setAtualizando] = useState(false);
  const [carregandoMais, setCarregandoMais] = useState(false);
  const [erro, setErro]             = useState<string | null>(null);

  const buscarDados = useCallback((resetar = true) => {
    if (resetar) {
      setCarregando(true);
      setErro(null);
    }
    api.get('/gamification/ranking?page=1')
      .then(res => {
        const data: RankingResponse = res.data;
        setRanking(data.ranking);
        setPropria(data.own_entry);
        setPagina(1);
        setHasMore(data.has_more);
      })
      .catch(() => setErro('Não foi possível carregar o ranking. Tente novamente.'))
      .finally(() => {
        setCarregando(false);
        setAtualizando(false);
      });
  }, []);

  const carregarMais = useCallback(() => {
    if (!hasMore || carregandoMais) return;
    const proximaPagina = pagina + 1;
    setCarregandoMais(true);
    api.get(`/gamification/ranking?page=${proximaPagina}`)
      .then(res => {
        const data: RankingResponse = res.data;
        setRanking(prev => [...prev, ...data.ranking]);
        setPagina(proximaPagina);
        setHasMore(data.has_more);
        // own_entry nao muda entre paginas
      })
      .catch(() => {}) // silencioso — itens existentes permanecem
      .finally(() => setCarregandoMais(false));
  }, [hasMore, carregandoMais, pagina]);

  const onRefresh = useCallback(() => {
    setAtualizando(true);
    buscarDados(false);
  }, [buscarDados]);

  useFocusEffect(
    useCallback(() => {
      buscarDados();
    }, [buscarDados])
  );

  return (
    <SafeAreaView style={{ flex: 1, backgroundColor: palette.primaryBlue }}>
      <Stack.Screen options={{ headerShown: false }} />

      {/* Header */}
      <YStack backgroundColor={palette.primaryBlue} pt="$6" pb="$4" px="$5">
        <Text fontSize={22} fontWeight="bold" color="white">
          Ranking
        </Text>
        <Text fontSize={12} color="rgba(255,255,255,0.75)" mt="$1">
          Top 100 alunos por XP
        </Text>
        {/* XP column header */}
        <XStack jc="flex-end" mt="$2" pr="$1">
          <Text fontSize={12} fontWeight="bold" color="rgba(255,255,255,0.75)">
            XP
          </Text>
        </XStack>
      </YStack>

      {/* Body */}
      <YStack f={1} backgroundColor={palette.darkBlue}>
        {carregando ? (
          <YStack f={1} ai="center" jc="center">
            <ActivityIndicator color={palette.primaryBlue} size="large" />
          </YStack>
        ) : erro ? (
          <YStack f={1} ai="center" jc="center" px="$6">
            <Text fontSize={15} color={palette.red} textAlign="center">
              {erro}
            </Text>
          </YStack>
        ) : ranking.length === 0 ? (
          <YStack f={1} ai="center" jc="center" px="$6" gap="$2">
            <Text fontSize={15} fontWeight="bold" color="white" textAlign="center">
              Nenhum dado disponível
            </Text>
            <Text fontSize={15} color="rgba(255,255,255,0.7)" textAlign="center">
              Complete um quiz para aparecer no ranking.
            </Text>
          </YStack>
        ) : (
          <View style={{ flex: 1 }}>
            <FlatList
              data={ranking}
              keyExtractor={(item) => String(item.id)}
              renderItem={({ item }) => <RankingRow item={item} />}
              onEndReached={carregarMais}
              onEndReachedThreshold={0.5}
              refreshControl={
                <RefreshControl
                  refreshing={atualizando}
                  onRefresh={onRefresh}
                  colors={[palette.primaryGreen]}
                />
              }
              ListFooterComponent={
                hasMore ? (
                  carregandoMais ? (
                    <YStack py="$4" ai="center">
                      <ActivityIndicator color={palette.primaryGreen} />
                    </YStack>
                  ) : (
                    <YStack px="$6" py="$4">
                      <AppButton
                        backgroundColor={palette.primaryGreen}
                        onPress={carregarMais}
                      >
                        Carregar mais
                      </AppButton>
                    </YStack>
                  )
                ) : null
              }
            />
            {/* Pinned own-row: always visible outside FlatList scroll */}
            {propria && (
              <YStack
                backgroundColor={palette.darkBlue}
                borderTopWidth={1}
                borderTopColor="rgba(255,255,255,0.15)"
              >
                <XStack px="$4" py="$2" ai="center">
                  <Text fontSize={12} color="rgba(255,255,255,0.6)" mr="$2">
                    Sua posição
                  </Text>
                </XStack>
                <RankingRow item={propria} isOwn />
              </YStack>
            )}
          </View>
        )}
      </YStack>
    </SafeAreaView>
  );
}
```
