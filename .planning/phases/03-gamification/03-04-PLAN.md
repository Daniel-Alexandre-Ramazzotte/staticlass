**2. Add calendar builder helper** — add this function before the component return, inside the component:

```typescript
// Constroi grid de 4 semanas (28 dias) a partir das datas praticadas
const buildCalendar = useCallback((historico: { criado_em: string }[]): (boolean | null)[][] => {
  const hoje = new Date();
  hoje.setHours(0, 0, 0, 0);
  // Conjunto de datas praticadas (YYYY-MM-DD)
  const praticados = new Set(
    historico.map(h => h.criado_em.slice(0, 10))
  );
  const grid: (boolean | null)[][] = [];
  // 4 semanas, a mais recente embaixo (linha 3 = semana atual)
  // Cada linha começa no domingo da semana correspondente
  for (let semana = 3; semana >= 0; semana--) {
    const linha: (boolean | null)[] = [];
    for (let dia = 0; dia < 7; dia++) {
      const d = new Date(hoje);
      const diaDaSemanaHoje = hoje.getDay(); // 0=dom
      const offset = semana * 7 + (diaDaSemanaHoje - dia);
      d.setDate(hoje.getDate() - offset);
      const key = d.toISOString().slice(0, 10);
      if (d > hoje) {
        linha.push(null); // futuro
      } else if (d.getTime() === hoje.getTime() && !praticados.has(key)) {
        linha.push(null); // hoje, ainda nao praticou
      } else {
        linha.push(praticados.has(key));
      }
    }
    grid.push(linha);
  }
  return grid;
}, []);
```

```typescript
useFocusEffect(
  useCallback(() => {
    if (!email) return;
    setDadosCarregando(true);

    Promise.all([
      api.get(`/users/profile/${email.trim()}`),
      api.get('/gamification/ranking?page=1'),
      api.get('/users/historico'),
    ])
      .then(([resPerfil, resRanking, resHistorico]) => {
        setXp(resPerfil.data?.xp ?? 0);
        setStreak(resPerfil.data?.streak ?? 0);
        const ownEntry = resRanking.data?.own_entry;
        setRankPosition(ownEntry ? ownEntry.posicao : null);
        const hist: { criado_em: string }[] = resHistorico.data ?? [];
        setCalendar(buildCalendar(hist));
      })
      .catch(() => {
        // Mantém valores anteriores em caso de erro (nao deixa tela em branco)
      })
      .finally(() => setDadosCarregando(false));
  }, [email, buildCalendar])
);
```
