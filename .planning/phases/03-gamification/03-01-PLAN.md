```python
from .. import db
from sqlalchemy import text
from datetime import date


def buscar_gamification_data(usuario_id: int) -> dict:
    """Retorna xp, streak e last_practice_date do usuario."""
    row = db.session.execute(
        text("""
            SELECT COALESCE(xp, 0) AS xp,
                   COALESCE(streak, 0) AS streak,
                   last_practice_date
            FROM users
            WHERE id = :id
        """),
        {"id": usuario_id},
    ).mappings().fetchone()
    if row is None:
        return {"xp": 0, "streak": 0, "last_practice_date": None}
    return {
        "xp": row["xp"],
        "streak": row["streak"],
        "last_practice_date": row["last_practice_date"],
    }


def update_xp_and_streak(usuario_id: int, xp_ganho: int, nova_streak: int, hoje: date) -> None:
    """Incrementa xp e atualiza streak e last_practice_date do usuario."""
    db.session.execute(
        text("""
            UPDATE users
            SET xp = COALESCE(xp, 0) + :xp_ganho,
                streak = :nova_streak,
                last_practice_date = :hoje
            WHERE id = :id
        """),
        {
            "xp_ganho": xp_ganho,
            "nova_streak": nova_streak,
            "hoje": hoje,
            "id": usuario_id,
        },
    )
    db.session.commit()


def get_ranking_page(limite: int, offset: int) -> list:
    """Retorna pagina do ranking com posicao calculada via ROW_NUMBER().
    ATENCAO: apenas PostgreSQL. Nao usar em testes com SQLite."""
    return db.session.execute(
        text("""
            SELECT
                id,
                name,
                COALESCE(xp, 0) AS xp,
                ROW_NUMBER() OVER (ORDER BY COALESCE(xp, 0) DESC) AS posicao
            FROM users
            WHERE role = 'aluno' AND active = TRUE
            ORDER BY xp DESC
            LIMIT :limite OFFSET :offset
        """),
        {"limite": limite, "offset": offset},
    ).mappings().all()


def get_own_rank(usuario_id: int) -> dict | None:
    """Retorna posicao e xp do proprio usuario no ranking global.
    ATENCAO: apenas PostgreSQL. Nao usar em testes com SQLite."""
    row = db.session.execute(
        text("""
            SELECT posicao, xp, name
            FROM (
                SELECT
                    id,
                    name,
                    COALESCE(xp, 0) AS xp,
                    ROW_NUMBER() OVER (ORDER BY COALESCE(xp, 0) DESC) AS posicao
                FROM users
                WHERE role = 'aluno' AND active = TRUE
            ) ranked
            WHERE id = :id
        """),
        {"id": usuario_id},
    ).mappings().fetchone()
    if row is None:
        return None
    return {"posicao": int(row["posicao"]), "xp": int(row["xp"]), "name": row["name"]}
```

```python
from datetime import date
from ..repositories.gamification_repository import (
    buscar_gamification_data,
    update_xp_and_streak,
)
from ..repositories.resultado_repository import salvar_resultado


def calcular_nova_streak(streak_atual: int, last_practice_date) -> int:
    """Calcula nova streak com base na data do ultimo treino.
    Regra: calendario diario (nao janela de 24h).
      - None (primeiro acesso): retorna 1
      - Mesmo dia (delta=0): mantém streak_atual
      - Dia consecutivo (delta=1): streak_atual + 1
      - Pulou dia(s) (delta>=2): reseta para 1 (hoje conta)
    """
    hoje = date.today()
    if last_practice_date is None:
        return 1
    delta = (hoje - last_practice_date).days
    if delta == 0:
        return streak_atual
    elif delta == 1:
        return streak_atual + 1
    else:
        return 1


def calcular_xp(acertos: int, streak_atual: int) -> tuple[int, float]:
    """Calcula XP ganho na sessao aplicando multiplicador de streak.
    Formula (D-01 a D-04):
      base = acertos * 10 + 20            (por-acerto + bonus de conclusao)
      streak >= 7 -> 1.5x
      streak >= 3 -> 1.25x
      else        -> 1.0x
      xp_ganho = int(base * multiplier)  (truncamento inteiro)
    Exemplo: 3 acertos, streak=7 -> (30+20)*1.5 = 75 XP
    """
    base = acertos * 10 + 20
    if streak_atual >= 7:
        multiplier = 1.5
    elif streak_atual >= 3:
        multiplier = 1.25
    else:
        multiplier = 1.0
    return int(base * multiplier), multiplier


def record_session_service(usuario_id: int, dados: dict) -> tuple[dict, int]:
    """Registra resultado do quiz, calcula XP com multiplicador e atualiza streak.

    Body esperado: { acertos: int, total: int, capitulo_id?: int, dificuldade?: int }
    Retorna: { message, xp_ganho, streak, multiplier } com status 201.

    Seguranca: usuario_id vem do JWT (get_jwt_identity()), nunca do body.
    Validacao: acertos e total sao int nao-negativos; acertos <= total.
    """
    acertos     = dados.get("acertos")
    total       = dados.get("total")
    capitulo_id = dados.get("capitulo_id")
    dificuldade = dados.get("dificuldade")

    if acertos is None or total is None:
        return {"error": "campos 'acertos' e 'total' sao obrigatorios"}, 400
    try:
        acertos = int(acertos)
        total   = int(total)
    except (TypeError, ValueError):
        return {"error": "acertos e total devem ser inteiros"}, 400
    if acertos < 0 or total < 0:
        return {"error": "acertos e total nao podem ser negativos"}, 400
    if acertos > total:
        return {"error": "acertos nao pode ser maior que total"}, 400

    try:
        # 1. Salva o resultado do quiz em quiz_resultados
        salvar_resultado(usuario_id, acertos, total, capitulo_id, dificuldade)

        # 2. Busca dados de gamificacao atuais do usuario
        dados_gam    = buscar_gamification_data(usuario_id)
        streak_atual = dados_gam["streak"] or 0
        last_practice = dados_gam["last_practice_date"]

        # 3. Calcula nova streak e XP
        nova_streak          = calcular_nova_streak(streak_atual, last_practice)
        xp_ganho, multiplier = calcular_xp(acertos, nova_streak)

        # 4. Persiste XP e streak
        update_xp_and_streak(usuario_id, xp_ganho, nova_streak, date.today())

        return {
            "message":    "sessao registrada",
            "xp_ganho":   xp_ganho,
            "streak":     nova_streak,
            "multiplier": multiplier,
        }, 201

    except Exception as e:
        return {"error": f"erro ao registrar sessao: {e}"}, 500
```
